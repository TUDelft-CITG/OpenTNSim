41 def generate(self, environment, vessel_name, fleet_distribution=None, scenario=None): 

42 """ Generate a vessel """ 

43 

44 if fleet_distribution == None: 

45 vessel_info = self.vessel_database.sample( 

46 n=1, random_state=int(1000 * random.random()) 

47 ) 

48 else: 

49 vessel_info = self.vessel_database.sample( 

50 n=1, weights=fleet_distribution,random_state=int(1000 * random.random()) 

51 ) 

52 vessel_data = {} 

53 

54 vessel_data["env"] = environment 

55 vessel_data["name"] = vessel_name 

56 

57 if scenario: 

58 vessel_info = vessel_info[vessel_info["scenario"] == scenario] 

59 

60 for key in vessel_info: 

61 if key == "vessel_id": 

62 vessel_data["id"] = vessel_info[key].values[0] 

63 else: 

64 vessel_data[key] = vessel_info[key].values[0] 

65 

66 if self.loaded == True: 

67 vessel_data["level"] = vessel_data["capacity"] 

68 elif self.loaded == "Random": 

69 if random.random() < 0.5: 

70 vessel_data["level"] = vessel_data["capacity"] 

71 else: 

72 vessel_data["level"] = 0

78 def arrival_process( 

79 self,

82 destination, 

83 arrival_distribution, 

84 scenario, 

85 arrival_process, 

86 fleet_distribution,

88 """  

89 Make arrival process 

90  

91 environment: simpy environment 

92 arrival_distribution: specify the distribution from which vessels are generated, int or list 

93 arrival_process: process of arrivals 

94 """ 

95 

96 # Create an array with inter-arrival times -- average number of seconds between arrivals 

97 if type(arrival_distribution) == int or type(arrival_distribution) == float: 

98 self.inter_arrival_times = [3600 / arrival_distribution] * 24 

99 

100 elif type(arrival_distribution) == list and len(arrival_distribution) == 24: 

101 self.inter_arrival_times = [3600 / n for n in arrival_distribution] 

102 

103 elif type(arrival_distribution) == list: 

104 raise ValueError( 

105 "List should contain an average number of vessels per hour for an entire day: 24 entries." 

106 ) 

107 

108 else: 

109 raise ValueError( 

110 "Specify an arrival distribution: type Integer or type List." 

111 ) 

112 

113 while True: 

114 

115 # Check simulation time 

116 inter_arrival = self.inter_arrival_times[ 

117 datetime.datetime.fromtimestamp(environment.now).hour 

118 ] 

119 

120 # In the case of a Markovian arrival process 

121 if arrival_process == "Markovian": 

122 

123 # Make a timestep based on the poisson process 

124 yield environment.timeout(random.expovariate(1 / inter_arrival)) 

125 

126 elif arrival_process == "Uniform": 

127 

128 # Make a timestep based on uniform arrivals 

129 yield environment.timeout(inter_arrival) 

130 

131 else: 

132 raise ValueError( 

133 "No other arrival processes are yet defined. You can add them to transport_network_analysis/vessel_generator.py." 

134 ) 

135 

136 # Create a vessel 

137 vessel = self.generate(environment, "Vessel", fleet_distribution, scenario) 

138 vessel.output = {} 

139 core.Output.vessel_dependent_output(vessel) 

140 vessel.env = environment 

141 vessel.route = nx.dijkstra_path(environment.FG, origin, destination) 

142 vessel.geometry = nx.get_node_attributes(environment.FG, "geometry")[ 

143 vessel.route[0] 

144 ] 

145 environment.vessels.append(vessel) 

146 # Move on path 

147 process = environment.process(vessel.move()) 

148 vessel.process = process
